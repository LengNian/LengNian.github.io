def DeepFool(image, label, net, device, num_classes=10, max_iter=50, overshoot=0.02):

    # f_imageä¸ºç»è¿‡æ¨¡å‹çš„å¯¹å„ä¸ªç±»åˆ«çš„é¢„æµ‹

    f_image = net(image).data.cpu().numpy().flatten()  # shape:[1, 10] -- [10]


    # argsort(æ’åº,è¿”å›åœ¨åŸæ•°æ®ä¸­çš„ç´¢å¼•ï¼Œä»å°åˆ°å¤§æ’,ä½¿ç”¨[::-1]å˜ä¸ºäº†ä»å¤§åˆ°å°)
    I = (np.array(f_image)).flatten().argsort()[::-1]

    I = I[0:num_classes]
    # æ¨¡å‹å®é™…é¢„æµ‹çš„æ ‡ç­¾,æˆ‘è¿™é‡Œæ”¹ä¸ºè¯¥æ•°æ®çš„çœŸå®æ ‡ç­¾
    # label = I[0]

    # è·å–è¾“å…¥å›¾åƒçš„å½¢çŠ¶
    # æºä»£ç è¾“å…¥imageæ˜¯H*W*C,æˆ‘è¿™é‡Œéœ€è¦é™ç»´
    # input_shape = image.cpu().numpy().shape
    input_shape = image.squeeze(dim=0).cpu().numpy().shape
    pert_image = copy.deepcopy(image)

    w = np.zeros(input_shape)  # shape[1, 28, 28]
    r_tot = np.zeros(input_shape) # shape[1, 28, 28]
    
    loop_i = 0
    # æºä»£ç è¿™é‡Œåˆå¢åŠ ä¸€ä¸ªç»´åº¦ï¼Œæˆ‘åªéœ€è¦ç›´æ¥æ·±æ‹·è´åŸå§‹å›¾åƒå³å¯
    # x = Variable(pert_image[None, :], requires_grad=True)
    x = torch.Tensor(pert_image)
    x.requires_grad = True

    fs = net(x)  # fs shape: [1, 10]
    # fsè¿™é‡Œçš„shapeæ˜¯[1,10],æ‰€ä»¥è¿™é‡Œä½¿ç”¨fs[0, I[k]],ä¹Ÿå¯ä»¥æŠŠfsé™ç»´æˆ[10]
    # è¿™æ˜¯æŠŠé¢„æµ‹çš„æ¦‚ç‡å€¼ä»å¤§åˆ°å°æ”¾åˆ°fs_listä¸­äº†
    fs_list = [fs[0, I[k]] for k in range(10)]

    # k_iè¡¨ç¤ºæ·»åŠ æ‰°åŠ¨ä»¥åé¢„æµ‹çš„ç±»åˆ«,è¿™é‡Œå¯¹k_iè¿›è¡Œåˆå§‹åŒ–
    k_i = label

    while k_i == label and loop_i < max_iter:
        # pertç”¨äºå­˜å‚¨æœ€å°çš„æ‰°åŠ¨
        pert = np.inf
        # è®¡ç®—è¯¥é¢„æµ‹æ ‡ç­¾åœ¨è¾“å…¥æ•°æ®xå¤„çš„æ¢¯åº¦
        fs[0, I[0]].backward(retain_graph=True)
        grad_orig = x.grad.data.cpu().numpy().copy()

        for k in range(1, 10):
            # å°†æ¨¡å‹ä¸­æ‰€æœ‰å‚æ•°çš„æ¢¯åº¦æ¸…é›¶
            zero_gradients(x)
            # x.grad.zero_()

            # è®¡ç®—ç¬¬kä¸ªæ ‡ç­¾åœ¨è¾“å…¥æ•°æ®xå¤„çš„æ¢¯åº¦
            fs[0, I[k]].backward(retain_graph=True)
            cur_grad = x.grad.data.cpu().numpy().copy()

            # è®¡ç®—ç¬¬kä¸ªç±»åˆ«å’Œåº”é¢„æµ‹ç±»åˆ«åœ¨xæ¢¯åº¦çš„å·®å¼‚,ä¹Ÿæ˜¯æ‰°åŠ¨åº”è¯¥æ·»åŠ çš„æ–¹å‘
            w_k = cur_grad - grad_orig
            f_k = (fs[0, I[k]] - fs[0, I[0]]).data.cpu().numpy()

            # è®¡ç®—äº†æ¢¯åº¦å‘é‡w_kçš„L2èŒƒæ•°,pert_kè¶Šå°ï¼Œæ‰°åŠ¨è¶Šå°ï¼Œæ”»å‡»è¶Šæœ‰æ•ˆ
            pert_k = abs(f_k) / np.linalg.norm(w_k.flatten())

            # é€‰æ‹©æœ€å°æ‰°åŠ¨è¿›è¡Œä¿å­˜
            if pert_k < pert:
                # éœ€è¦ç§»åŠ¨çš„è·ç¦»
                pert = pert_k
                # æ‰°åŠ¨æ–¹å‘
                w = w_k
        
        r_i = (pert + 1e-4) * w / np.linalg.norm(w)
        # ç´¯åŠ åˆ°æ€»æ‰°åŠ¨ä¸Š
        r_tot = np.float32(r_tot + r_i)

        pert_image = image + (1 + overshoot) * torch.from_numpy(r_tot).cuda()

        # x = Variable(pert_image, requires_grad=True)
        x = torch.Tensor(pert_image)
        x.requires_grad = True

        fs = net(x)
        # k_iè¡¨ç¤ºé¢„æµ‹æ ‡ç­¾
        k_i = np.argmax(fs.data.cpu().numpy().flatten())

        loop_i += 1
        
    # æ”¾å¤§æ‰°åŠ¨
    r_tot = (1+overshoot) * r_tot

    return r_tot, k_i, pert_image

class LeNet(nn.Module):
    def __init__(self):
        super(LeNet, self).__init__()

        # å·ç§¯å±‚
        self.conv = nn.Sequential(
            nn.Conv2d(in_channels=1, out_channels=6, kernel_size=5, padding=2),
            nn.ReLU(),
            nn.MaxPool2d(kernel_size=2, stride=2),
            nn.Conv2d(in_channels=6, out_channels=16, kernel_size=5),
            nn.ReLU(),
            nn.MaxPool2d(kernel_size=2, stride=2)
        )

        # å…¨è¿æ¥å±‚
        self.fc = nn.Sequential(
            nn.Linear(in_features=16 * 5 * 5, out_features=120),
            nn.ReLU(),
            nn.Linear(in_features=120, out_features=84),
            nn.ReLU(),
            nn.Linear(in_features=84, out_features=10)
        )

    def forward(self, img):
        img = self.conv(img)
        img = img.view(img.size(0), -1)
        out = self.fc(img)
        # out = self.softmax(out)
        
        return out

    f_image = net(image).data.cpu().numpy().flatten()  # shape:[1, 10] -- [10]


    # argsort(æ’åº,è¿”å›åœ¨åŸæ•°æ®ä¸­çš„ç´¢å¼•ï¼Œä»å°åˆ°å¤§æ’,ä½¿ç”¨[::-1]å˜ä¸ºäº†ä»å¤§åˆ°å°)
    I = (np.array(f_image)).flatten().argsort()[::-1]

    # label = I[0]

    # input_shape = image.cpu().numpy().shape
    input_shape = image.squeeze(dim=0).cpu().numpy().shape

    loop_i = 0
    # æºä»£ç è¿™é‡Œåˆå¢åŠ ä¸€ä¸ªç»´åº¦ï¼Œæˆ‘åªéœ€è¦ç›´æ¥æ·±æ‹·è´åŸå§‹å›¾åƒå³å¯
    # x = Variable(pert_image[None, :], requires_grad=True)

            zero_gradients(x)

            fs[0, I[k]].backward(retain_graph=True)

            # è®¡ç®—ç¬¬kä¸ªç±»åˆ«å’Œåº”é¢„æµ‹ç±»åˆ«åœ¨xæ¢¯åº¦çš„å·®å¼‚,ä¹Ÿæ˜¯æ‰°åŠ¨åº”è¯¥æ·»åŠ çš„æ–¹å‘

            f_k = (fs[0, I[k]] - fs[0, I[0]]).data.cpu().numpy()

            # è®¡ç®—äº†æ¢¯åº¦å‘é‡w_kçš„L2èŒƒæ•°,pert_kè¶Šå°ï¼Œæ‰°åŠ¨è¶Šå°ï¼Œæ”»å‡»è¶Šæœ‰æ•ˆ
            pert_k = abs(f_k) / np.linalg.norm(w_k.flatten())

            # é€‰æ‹©æœ€å°æ‰°åŠ¨è¿›è¡Œä¿å­˜
            if pert_k < pert:

                # æ‰°åŠ¨æ–¹å‘
                w = w_k
        
        r_i = (pert + 1e-4) * w / np.linalg.norm(w)
        # ç´¯åŠ åˆ°æ€»æ‰°åŠ¨ä¸Š
        r_tot = np.float32(r_tot + r_i)

        x = torch.Tensor(pert_image)

        loop_i += 1
        
    # æ”¾å¤§æ‰°åŠ¨
    r_tot = (1+overshoot) * r_tot

    return r_tot, k_i, pert_image
